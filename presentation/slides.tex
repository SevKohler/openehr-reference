\section{What is a clinical reference?}

\begin{frame}{What is a clinical reference?}
  \begin{itemize}
    \item Clinical data points to other clinical data
    \item Procedure $\rightarrow$ reason, Device $\rightarrow$ entry, Composition $\rightarrow$ encounter
    \item Target may be inside the same EHR or in an external system
    \item A \textbf{clinical reference} is that typed pointer
    \item  How do you model it so it is typed, constrainable, and interoperable?
  \end{itemize}

\end{frame}

\begin{frame}{A concrete example}
  \begin{itemize}
    \item \texttt{Procedure.reason} should accept a typed reference or a static code
    \item Linked \texttt{problem\_diagnosis} exists: point to it, surface diagnosis code inline
    \item Nothing linked: fall back to a coded value from a value set
    \item Both cases belong in the same field, constrained by the archetype
    \item No single mechanism covers this. Modelers duplicate the value, use a free-text \texttt{LINK} nobody queries, or leave the field disconnected.
  \end{itemize}
\end{frame}

\section{How FHIR solves it}

\begin{frame}{FHIR: \texttt{Reference}}
  \begin{itemize}
    \item First-class data type, used explicitly per resource field
    \item Carries: reference string, type, logical identifier, display label
  \end{itemize}
  \begin{figure}
    \includegraphics[width=1\textwidth]{../Reference.png}
  \end{figure}
\end{frame}

\begin{frame}{FHIR: \texttt{CodeableReference}}
  \begin{itemize}
    \item Holds a coded concept, a \texttt{Reference}, or both in one field
    \item \texttt{Procedure.reason}: free code when nothing linked, reference when it exists
  \end{itemize}
  \begin{figure}
    \includegraphics[width=1\textwidth]{../CodeableReference.png}
  \end{figure}
\end{frame}

\begin{frame}{What FHIR gets right}
  \begin{itemize}
    \item Explicitly typed, not free text
    \item Pointer and coded fallback in the same field
    \item Constrainable per field, not attachable everywhere
    \item Fixed structure enforces interoperability
  \end{itemize}
\end{frame}

\section{How openEHR handles it today}

\begin{frame}{openEHR: \texttt{LINK}}
  \begin{itemize}
    \item Available on every \texttt{LOCATABLE}
    \item Carries meaning, type (free string), and target URI
  \end{itemize}
\end{frame}

\begin{frame}{openEHR: external references}
  \begin{itemize}
    \item Modeled separately via \texttt{PARTY\_REF}, \texttt{OBJECT\_REF} and related classes
    \item A different mechanism from \texttt{LINK}
  \end{itemize}

  \begin{figure}
    \includegraphics[width=1\textwidth]{../PARTY.png}
  \end{figure}
\end{frame}

\begin{frame}{Problem 5: post-coordinated terms}
  \begin{itemize}
    \item Neither FHIR nor openEHR has a standard way to represent post-coordinated expressions
    \item Example: "Malignant neoplasm of right testis" composed from finding + laterality + site
    \item FHIR has no dedicated structure for it
    \item openEHR has no dedicated structure for it
    \item Currently handled ad-hoc or not at all
  \end{itemize}
  \begin{alertblock}{Open in both}
    Post-coordination is an unsolved problem across standards.
  \end{alertblock}
\end{frame}

\section{The problems}

\begin{frame}{Problems: \texttt{LINK}}
  \begin{itemize}
    \item Attachable to any \texttt{LOCATABLE}, cannot be scoped to a field
    \item \texttt{type} is a free string, no terminology binding or validation
    \item Cannot be placed inside \texttt{ELEMENT.value}
    \item No shared abstraction with external references (\texttt{OBJECT\_REF}, \texttt{PARTY\_REF})
    \item Every team uses it differently, cross-system querying breaks
  \end{itemize}
\end{frame}

\begin{frame}{Problems: missing capabilities}
  \begin{itemize}
    \item No way to express a reference and a coded fallback in one field
    \item External references are RM-level only, not usable in archetypes
    \item No mechanism to surface values from a linked target inline
    \item Duplicating values is the only workaround, duplicates go stale
    \item Post-coordinated terms have no dedicated structure in either standard
  \end{itemize}
\end{frame}
