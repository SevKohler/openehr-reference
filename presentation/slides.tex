\section{What is a clinical reference?}

\begin{frame}{What is a clinical reference?}
  \begin{itemize}
    \item Clinical data points to other clinical data
    \item Target may be inside the same EHR or in an external system
    \item A \textbf{clinical reference} is that typed pointer
    \item  How do you model it so it is typed, constrainable, and interoperable?
  \end{itemize}

\end{frame}

\begin{frame}{A concrete example}
  \begin{itemize}
    \item \texttt{Procedure.reason} should accept a typed reference or a static code
    \item Linked \texttt{problem\_diagnosis} exists: point to it, surface diagnosis code inline
    \item Nothing linked: fall back to a coded value from a value set
    \item Both cases belong in the same field, constrained by the archetype
  \end{itemize}
\end{frame}

\section{How FHIR solves it}

\begin{frame}{FHIR: \texttt{Reference}}
  \begin{itemize}
    \item used explicitly per resource field
    \item reference string, type, logical identifier, display label
  \end{itemize}
  \begin{figure}
    \includegraphics[width=1\textwidth]{../Reference.png}
  \end{figure}
\end{frame}

\begin{frame}{FHIR: \texttt{CodeableReference}}
  \begin{itemize}
    \item Holds a coded concept, a \texttt{Reference}, or both in one field
    \item \texttt{Procedure.reason}: free code when nothing linked, reference when it exists
  \end{itemize}
  \begin{figure}
    \includegraphics[width=1\textwidth]{../CodeableReference.png}
  \end{figure}
\end{frame}

\begin{frame}{What FHIR gets right}
  \begin{itemize}
    \item Explicitly typed, not free text
    \item Pointer and code
    \item not attachable everywhere, explicit modeled
    \item Fixed structure enforces interoperability
  \end{itemize}
\end{frame}

\section{How openEHR handles it today}

\begin{frame}{openEHR: \texttt{LINK}}
  \begin{itemize}
    \item Available on every \texttt{LOCATABLE}
    \item Carries meaning, type (free string), and target URI
    \item Only mechanism for internal references
    \item no archetype-level control cannot be constrained
  \end{itemize}
  \begin{figure}
    \includegraphics[width=\textwidth]{../LINKS.png}
  \end{figure}
\end{frame}

\begin{frame}{openEHR: DV\_EHR\_URI and DV\_URI}
  \begin{itemize}
    \item data value can be modeled explicitly
    \item cannot be constrained to e.g. just linking EVALUATION
  \end{itemize}

  \begin{figure}
    \includegraphics[width=1\textwidth]{../DV_EHR_URI.png}
  \end{figure}
\end{frame}

\begin{frame}{openEHR: external references}
  \begin{itemize}
    \item Modeled explicitly, also part of the RM classes via \texttt{DV\_URI} \texttt{PARTY\_REF}, \texttt{OBJECT\_REF} and related classes
    \item A different mechanism from \texttt{LINK}
  \end{itemize}

  \begin{figure}

\end{frame}

\section{The problems}

\begin{frame}{Problems: \texttt{LINK}}
  \begin{itemize}
    \item Attachable to any \texttt{LOCATABLE}, cannot be scoped to a field
    \item \texttt{type} is a free string, no terminology binding or validation
    \item No shared abstraction with external references (\texttt{OBJECT\_REF}, \texttt{PARTY\_REF})
    \item You cannot say what to expect from the link target
    \item There also explicit modeled references via DV\_EHR\_URI
    \item Every team uses it differently, cross-system querying breaks
  \end{itemize}
\end{frame}

\begin{frame}{Problems: post-coordinated terms}
  \begin{itemize}
    \item Neither FHIR nor openEHR has a standard way to represent post-coordinated expressions
    \item Example: "Radiation therapy for kidney cancer" composed from treatment + indication
    \item FHIR has no dedicated structure for it
    \item openEHR has no dedicated structure for it
    \item Currently handled ad-hoc or not at all
    \item necessary ? its colliding existing fields 
  \end{itemize}
\end{frame}

\begin{frame}{Summary}
  \begin{itemize}
    \item No way to express a reference and a coded fallback in one field
    \item Mix between explicit modeled, RM defined, implicit via \texttt{LINK}
    \item No mechanism to surface values from a linked target inline
    \item Duplicating values is the only workaround, duplicates go stale
    \item Post-coordinated terms have no dedicated structure in either standard
  \end{itemize}
\end{frame}

\section{Proposal}

\begin{frame}{Proposal: \texttt{DV\_REFERENCE}}
  \begin{itemize}
    \item New \texttt{DATA\_VALUE} subclass, usable inside \texttt{ELEMENT.value}
    \item Unifies internal and external references in one type
    \item \texttt{internal\_ref}: pointer to a \texttt{LOCATABLE} inside the EHR
    \item \texttt{external\_ref}: pointer to an entity outside openEHR
    \item \texttt{proxy}: values surfaced from the target or hardcoded as fallback
    \item Coded fallback: same field accepts \texttt{DV\_CODED\_TEXT} when nothing is linked
  \end{itemize}
\end{frame}

\begin{frame}{\texttt{DV\_REFERENCE}}
  \begin{figure}
    \includegraphics[width=\textwidth]{../DV_REFERENCE.png}
  \end{figure}
\end{frame}

\begin{frame}{Proxy values}
  \begin{itemize}
    \item \texttt{proxy} holds a list of \texttt{PROXY\_VALUE} entries
    \item Each entry carries a path relative to \texttt{internal\_ref} and a resolved value
    \item Path resolved at recording time, value stored alongside the reference
    \item No duplication: source is authoritative, proxy is a snapshot
    \item Without a link: proxy carries hardcoded values set by the modeler
  \end{itemize}
\end{frame}

\begin{frame}{\texttt{PROXY\_VALUES}}
  \begin{figure}
    \includegraphics[width=\textwidth]{../proxy_values.png}
  \end{figure}
\end{frame}

\begin{frame}{Example: \texttt{Procedure.reason}}
  \begin{itemize}
    \item \texttt{ELEMENT.value} constrained to \texttt{DV\_REFERENCE}
    \item \texttt{internal\_ref} points to a \texttt{problem\_diagnosis}
    \item \texttt{proxy} surfaces the diagnosis code from the linked archetype
    \item If no diagnosis is linked: field falls back to a coded value from the reason value set
    \item One field, one constraint, two valid states
  \end{itemize}
\end{frame}

\begin{frame}{Two kinds of \texttt{DV\_REFERENCE}}
  \begin{columns}
    \column{0.48\textwidth}
      \textbf{RM-level (hardcoded)}
      \begin{itemize}
        \item Typed attribute baked into the RM class
        \item Example: \texttt{EVENT\_CONTEXT.encounter}
        \item Scope and type fixed by the spec, no archetype control
        \item Reserved for structural slots every composition shares
      \end{itemize}
    \column{0.48\textwidth}
      \textbf{Archetype-level (explicit)}
      \begin{itemize}
        \item \texttt{DV\_REFERENCE} inside \texttt{ELEMENT.value}
        \item Defined and constrained by the modeler per archetype
        \item Target type, archetype id, proxy paths all constrainable
        \item The normal case for all clinical references
      \end{itemize}
  \end{columns}

\end{frame}


\begin{frame}{Post-coordinated terms}
  \begin{itemize}
    \item \texttt{PROXY\_EXPRESSION} extends \texttt{PROXY\_BASE} for composed concepts
    \item Holds a list of \texttt{EXPRESSION\_DEFINITION} entries
    \item Each component is a \texttt{(term, path)} pair: one axis of the definition
    \item Example: treatment + indication compose "Radiation therapy for kidney cancer"
    \item Single assembled \texttt{value} inherited from \texttt{PROXY\_BASE}
  \end{itemize}
\end{frame}

\begin{frame}{FHIR alignment: \texttt{Reference}}
  \begin{itemize}
    \item FHIR \texttt{Reference}: \texttt{reference} (URL) + \texttt{identifier} (logical) + \texttt{display}
    \item \texttt{internal\_ref} $\leftrightarrow$ FHIR \texttt{reference}: resolvable pointer to the target
    \item \texttt{external\_ref} $\leftrightarrow$ FHIR \texttt{identifier}: logical id for an external entity
    \item \texttt{display} maps directly
    \item FHIR allows both at once; \texttt{DV\_REFERENCE} enforces mutual exclusion by invariant
  \end{itemize}
\end{frame}

\begin{frame}{FHIR alignment: \texttt{CodeableReference}}
  \begin{itemize}
    \item \texttt{CodeableReference}: code \textbf{or} \texttt{Reference} in one field
    \item Coded case: \texttt{proxy} holds a hardcoded \texttt{DV\_CODED\_TEXT}, no link set
    \item Reference case: \texttt{internal\_ref}/\texttt{external\_ref} set, \texttt{proxy} resolved from target
    \item Identical to \texttt{Procedure.reason}: same field, two valid states
  \end{itemize}
\end{frame}

\section{Open points}

\begin{frame}{Open points}
  \begin{itemize}
    \item \texttt{EHR\_PATH} has no positional predicate for \texttt{0..*} nodes, paths are ambiguous
    \item Do we want post\_coordinated terms ? shouldnt these just be proxied
    \item how does the modeling look ?
    \item Should \texttt{shared\_link} forbid \texttt{proxy} via a stricter subtype of \texttt{DV\_REFERENCE}?
    \item Resolution behaviour when linked target no longer satisfies local constraint
    \item Legacy data: deprecate \texttt{LINK} ? how to handle existing \texttt{OBJECT\_REF} / \texttt{PARTY\_REF}
  \end{itemize}
\end{frame}
